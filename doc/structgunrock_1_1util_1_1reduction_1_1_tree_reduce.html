template<int LOG_CTA_THREADS, bool ALL_RETURN>
struct gunrock::util::reduction::TreeReduce< LOG_CTA_THREADS, ALL_RETURN >

Perform LOG_CTA_THREADS steps of binary tree reduction, each thread contributes one reduction partial.
