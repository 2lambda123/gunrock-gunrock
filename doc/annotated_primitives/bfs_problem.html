#

MARK_PREDECESSOR would be true when algorithm needs to know the predecessor node IDs for a newly formed frontier during a certain step.

    static const bool MARK_PREDECESSORS     = _MARK_PREDECESSORS;
#

ENABLE_IDEMPOTENCE would be true when the operation performed in parallel for all neighbor nodes/edges is idempotent, meaning data race is benign.

    static const bool ENABLE_IDEMPOTENCE    = _ENABLE_IDEMPOTENCE;

    cudaError_t Init(
            const Csr<VertexId, Value, SizeT> &graph)
    {
        cudaError_t retval = cudaSuccess;
        if (retval = util::GRError(ProblemBase::Init(graph))) break;
        if (retval = util::GRError(GPUMalloc(data_slices[0]->d_labels, nodes))) break;

        return retval;
